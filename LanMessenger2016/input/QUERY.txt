CREATE TABLE IF NOT EXISTS USER (ID INT PRIMARY KEY  ASC, USER_NAME VARCHAR NOT NULL, PASSWORD VARCHAR NOT NULL, IP VARCHAR NOT NULL, EMAIL VARCHAR, VERIFIED BOOL, PHONE VARCHAR)

CREATE TABLE IF NOT EXISTS FRIENDLIST (ID INT PRIMARY KEY  ASC, USER_NAME VARCHAR NOT NULL, FRIEND_UNAME VARCHAR NOT NULL, LAST_COMMUNICATED DATE)


CREATE TABLE IF NOT EXISTS CHATHISTORY ( USER_NAME VARCHAR NOT NULL, FRIEND_UNAME VARCHAR NOT NULL, MSG VARCHAR NOT NULL, CHAT_DATE DATE, CHAT_TIME DATETIME)
